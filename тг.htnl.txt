    <script src="https://cdn.jsdelivr.net/npm/qrcode@1.5.3/build/qrcode.min.js"></script>
    <script>
        // Получаем параметры из URL
        const urlParams = new URLSearchParams(window.location.search);
        
        // Устанавливаем значения или используем значения по умолчанию
        const data = {
            fio: urlParams.get('fio') || 'Иванов Иван Иванович',
            phone: urlParams.get('phone') || '+7 (999) 123-45-67',
            tg: urlParams.get('tg') || 'https://t.me/ваш_канал'
        };
        
        // Обновляем данные на странице
        document.getElementById('fio').textContent = data.fio;
        document.getElementById('phone').textContent = data.phone;
        document.getElementById('tg-link').href = data.tg;
        
        // Форматируем текст для кнопки Telegram
        const tgUsername = data.tg.split('/').pop();
        document.getElementById('tg-text').textContent = `Перейти в @${tgUsername}`;
        
        // Генерация QR-кода с красивой рамкой
        const qrSize = 220;
        const qrOptions = {
            width: qrSize,
            height: qrSize,
            colorDark: "#0088cc",
            colorLight: "#ffffff",
            correctLevel: QRCode.CorrectLevel.H
        };
        
        // Создаем контейнер для QR-кода
        const qrContainer = document.getElementById('qrcode');
        
        // Генерируем QR-код
        QRCode.toCanvas(qrContainer, window.location.href, qrOptions, function(error, canvas) {
            if (error) {
                console.error(error);
                qrContainer.innerHTML = '<p style="color: #ff4444;">Ошибка генерации QR-кода</p>';
                return;
            }
            
            // Добавляем стили к canvas
            canvas.style.borderRadius = '10px';
            canvas.style.boxShadow = '0 5px 15px rgba(0,0,0,0.1)';
            
            // Добавляем логотип Telegram поверх QR-кода
            const ctx = canvas.getContext('2d');
            
            // Создаем иконку Telegram
            const iconSize = qrSize * 0.2;
            const iconX = (qrSize - iconSize) / 2;
            const iconY = (qrSize - iconSize) / 2;
            
            // Рисуем круг для иконки
            ctx.beginPath();
            ctx.arc(iconX + iconSize/2, iconY + iconSize/2, iconSize/2 + 2, 0, 2 * Math.PI);
            ctx.fillStyle = '#ffffff';
            ctx.fill();
            
            // Добавляем иконку Telegram (упрощенная версия)
            ctx.fillStyle = '#0088cc';
            ctx.beginPath();
            ctx.arc(iconX + iconSize/2, iconY + iconSize/2, iconSize/2 - 2, 0, 2 * Math.PI);
            ctx.fill();
            
            // Буква "T" для Telegram
            ctx.fillStyle = '#ffffff';
            ctx.font = `bold ${iconSize * 0.6}px Arial`;
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            ctx.fillText('T', iconX + iconSize/2, iconY + iconSize/2);
        });
        
        // Добавляем эффект при загрузке
        document.addEventListener('DOMContentLoaded', function() {
            document.body.style.opacity = '0';
            document.body.style.transition = 'opacity 0.5s ease';
            
            setTimeout(() => {
                document.body.style.opacity = '1';
            }, 100);
        });
    </script>
